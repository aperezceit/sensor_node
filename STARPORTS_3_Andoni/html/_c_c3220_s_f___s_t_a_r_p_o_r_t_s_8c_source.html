157  * This is the handler for the uDMA error interrupt.
158  */
159 static void dmaErrorFxn(uintptr_t arg)
160 {
161  int status = MAP_uDMAErrorStatusGet();
162  MAP_uDMAErrorStatusClear();
163 
164  /* Suppress unused variable warning */
165  (void)status;
166 
167  while (1);
168 }
