336 void ADXL355_Get_Accel_Frame(SPI_Handle spi, uint16_t Samples, int32_t *DataSensor) {
337 
338  int i=0;
339  int32_t Ax[512], Ay[512], Az[512];
340  int32_t xMean, yMean, zMean;
341  double xRms, yRms, zRms;
342 
343  xMean=0;
344  yMean=0;
345  zMean=0;
346  while (i<Samples) {
347  while(!ADXL355_Data_Rdy(spi)) {}
348  Ax[i] = ADXL355_XData(spi);
349  Ay[i] = ADXL355_YData(spi);
350  Az[i] = ADXL355_ZData(spi);
351  xMean += Ax[i];
352  yMean += Ay[i];
353  zMean += Az[i];
354  i++;
355  }
356 
357  xMean = xMean / Samples;
358  yMean = yMean / Samples;
359  zMean = zMean / Samples;
360 
361  xRms = 0.0;
362  yRms = 0.0;
363  zRms = 0.0;
364  for (i=0;i<Samples;i++) {
365  xRms += ( (Ax[i]-xMean)*(Ax[i]-xMean) );
366  yRms += ( (Ay[i]-yMean)*(Ay[i]-yMean) );
367  zRms += ( (Az[i]-zMean)*(Az[i]-zMean) );
368  }
369 
370  DataSensor[0] = xMean;
371  DataSensor[1] = yMean;
372  DataSensor[2] = zMean;
373  DataSensor[3] = (int32_t)sqrt(xRms/Samples);
374  DataSensor[4] = (int32_t)sqrt(yRms/Samples);
375  DataSensor[5] = (int32_t)sqrt(zRms/Samples);
376 }
